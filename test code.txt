
import smtplib
from email.mime.text import MIMEText
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email import encoders

# Set up the email details
sender_email = "your_email@gmail.com"
recipient_email = "recipient_email@example.com"
subject = "Attachment Test"
message = "Hello, please find the attached PDF."

# Create a message object
msg = MIMEMultipart()
msg['From'] = sender_email
msg['To'] = recipient_email
msg['Subject'] = subject

# Attach a PDF file
pdf_file_path = "path_to_your_pdf_file.pdf"  # Replace with the actual path to your PDF file

# Create a MIME attachment
attachment = open(pdf_file_path, "rb")
part = MIMEBase('application', 'octet-stream')
part.set_payload(attachment.read())
encoders.encode_base64(part)
part.add_header('Content-Disposition', f'attachment; filename="{pdf_file_path}"')

# Attach the PDF to the message
msg.attach(part)

# Add the message body (plain text)
msg.attach(MIMEText(message, 'plain'))

# Connect to the SMTP server and send the email
smtp_server = "smtp.gmail.com"  # Update with your SMTP server
smtp_port = 587  # Update with your SMTP port (587 for TLS/SSL)
smtp_username = "your_email@gmail.com"  # Your email address
smtp_password = "your_password"  # Your email password

try:
    server = smtplib.SMTP(smtp_server, smtp_port)
    server.starttls()
    server.login(smtp_username, smtp_password)
    server.sendmail(sender_email, recipient_email, msg.as_string())
    server.quit()
    print("Email sent successfully with attachment!")
except Exception as e:
    print("Email sending failed:", str(e))
